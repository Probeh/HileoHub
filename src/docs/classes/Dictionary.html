import { IKeyValue } from '@helpers/key-value.interface'

export interface IDictionary {
  // ======================================= //
  [key: string]: any[] | IKeyValue<string, object>[];
  // ======================================= //
  keys?: string[];
  values?: object[];
  pairs?: { key: string, value: IKeyValue<string, object> }[];
  // ======================================= //
}
export class Dictionary implements IDictionary {
  // ======================================= //
  [key: string]: any | IKeyValue<string, object>[];
  // ======================================= //
  public get keys() {
    return Object.keys(this);
  }
  public get values() {
    return Object.keys(this).map(key => this[key]);
  }
  public get pairs() {
    return Object.keys(this).map(key => { return { key: key, value: this[key] } });
  }
  // ======================================= //
  constructor(key: string, value?: IKeyValue<string, object>) {
    this.append(key, value);
  }
  // ======================================= //
  public append(key: string, value?: IKeyValue<string, object>): Dictionary {
    if (!this[key]) {
      this[key] = new Array<IKeyValue<string, object>>();
    }
    if (value) {
      (this[key] as IKeyValue<string, object>[]).push(value);
    }
    return this;
  }
  public remove(key: string, value?: IKeyValue<string, object>): Dictionary {
    this[key] = value ? (this[key] as []).splice((this[key] as IKeyValue<string, object>[])?.indexOf(value), 1) : [];
    return this;
  }
  public getEntries(key: string): IKeyValue<string, object>[] {
    return this[key];
  }
  public update(key: string, value: IKeyValue<string, object>): Dictionary {
    this[key] = value;
    return this;
  }
}
